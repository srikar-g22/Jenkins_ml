pipeline {
    agent any
    tools {
        docker {
            // specify the Docker image to use for the build
            install 'docker'
            // arguments to pass to the Docker container
            arguments '-u root'
        }
    }
    stages {
        stage('Build') {
            steps {
                sh 'pip install -r requirements.txt'
            }
        }
        stage('Train') {
            steps {
                sh 'python trainingML.py'
            }
        }  
    }
}

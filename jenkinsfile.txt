pipeline {
    agent any
    tools {
        dockerTool 'docker_jenkins'
    }
    stages {
        stage('Setup') {
            steps {
                script {
                    docker.image('python:3.8-slim').inside {
                        sh 'pip3 install -r requirements.txt'
                    }
                }
            }
        }
        stage('Train') {
            steps {
                script {
                    docker.image('python:3.8-slim').inside {
                        sh 'python trainingML.py'
                    }
                }
            }
        }  
    }
}
